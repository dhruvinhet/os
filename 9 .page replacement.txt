#include <stdio.h>

int main() {
    int frames, pageref[30], frame[10], last_used[10];
    int n, i, j, page_faults = 0, flag, min, replace_index;

    printf("Enter number of frames: ");
    scanf("%d", &frames);

    printf("Enter length of page reference string (max size 30): ");
    scanf("%d", &n);
    
    printf("Enter the page reference string: ");
    for(i = 0; i < n; i++) {
        scanf("%d", &pageref[i]);
    }

    for(i = 0; i < frames; i++) {
        frame[i] = -1;
        last_used[i] = -1;
    }

    printf("\nPage Replacement Process (LRU):\n");

    for(i = 0; i < n; i++) {
        flag = 0;

        // Check if the page is already in memory (HIT)
        for(j = 0; j < frames; j++) {
            if(frame[j] == pageref[i]) {
                flag = 1;
                last_used[j] = i; // Update last used time
                printf("\nRequest %d: Page %d -> HIT", i + 1, pageref[i]);
                break;
            }
        }

        // If page not found (FAULT)
        if(flag == 0) {
            // Find an empty frame first
            replace_index = -1;
            for(j = 0; j < frames; j++) {
                if(frame[j] == -1) {
                    replace_index = j;
                    break;
                }
            }

            // If no empty frame, replace the least recently used one
            if(replace_index == -1) {
                min = 9999;
                for(j = 0; j < frames; j++) {
                    if(last_used[j] < min) {
                        min = last_used[j];
                        replace_index = j;
                    }
                }
            }

            // Replace the page
            frame[replace_index] = pageref[i];
            last_used[replace_index] = i;
            page_faults++;

            printf("\nRequest %d: Page %d -> FAULT, replaced at frame %d", i + 1, pageref[i], replace_index);
            printf("\nCurrent Frames: ");
            for(j = 0; j < frames; j++) {
                if(frame[j] != -1)
                    printf("%d ", frame[j]);
                else
                    printf("- ");
            }
        }
    }

    printf("\n\nTotal Page Faults = %d\n", page_faults);
    printf("Page Fault Rate = %.2f%%\n", ((float)page_faults / n) * 100);

    return 0;
}
