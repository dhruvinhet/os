#include <stdio.h>

int main() {
    int frames, pageref[30], frame[10];
    int n, i, j, k = 0, page_faults = 0, flag;

    printf("Enter number of frames: ");
    scanf("%d", &frames);

    printf("Enter length of page reference string (max size 30): ");
    scanf("%d", &n);
    
    printf("Enter the page reference string: ");
    for(i = 0; i < n; i++) {
        scanf("%d", &pageref[i]);
    }

    for(i = 0; i < frames; i++)
        frame[i] = -1;

    printf("\nPage Replacement Process (FIFO):\n");

    for(i = 0; i < n; i++) {
        flag = 0;

        for(j = 0; j < frames; j++) {
            if(frame[j] == pageref[i]) {
                flag = 1;
                printf("\nRequest %d: Page %d HIT", i + 1, pageref[i]);
                break;
            }
        }

        if(flag == 0) {
            frame[k] = pageref[i];
            printf("\nRequest %d: Page %d -> FAULT, replaced at frame %d", i + 1, pageref[i], k);
            k = (k + 1) % frames;
            page_faults++;

            printf("\nCurrent Frames: ");
            for(j = 0; j < frames; j++) {
                if(frame[j] != -1)
                    printf("%d ", frame[j]);
                else
                    printf("- ");
            }
        }
    }

    printf("\n\nTotal Page Faults = %d\n", page_faults);
    printf("Page Fault Rate = %.2f%%\n", ((float)page_faults / n) * 100);

    return 0;
}
