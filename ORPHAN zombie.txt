#include<stdio.h>
#include<stdlib.h>
#include<sys/wait.h>
#include<unistd.h>
#include<sys/types.h>

int main(){
    int pid = fork();
    if (pid >0){
        printf("parent process id is %d\n", getpid());
        printf("child process id %d\n", pid);
    } else if (pid == 0){
        sleep(1);
        printf("child process id is %d\n", getpid());
        printf("parent process id %d\n", getppid());
    }
    return 0;
}

#include<stdio.h>
#include<stdlib.h>
#include<sys/wait.h>
#include<unistd.h>
#include<sys/types.h>

void ascending_order(int arr[]){
    int i, j, temp;
    for (i = 0; arr[i] != '\0'; i++) {
        for (j = i + 1; arr[j] != '\0'; j++) {
            if (arr[i] > arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
}

void descending_order(int arr[]){
    int i, j, temp;
    for (i = 0; arr[i] != '\0'; i++) {
        for (j = i + 1; arr[j] != '\0'; j++) {
            if (arr[i] < arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }
}

int main(){
    int arr[5] = {2,1,5,7,8};
    int num = fork();
    if (num < 0)
        printf("Fork unsuccessful");
    else if (num == 0){
        ascending_order(arr);
        printf("Sorted array in ascending order: ");
        for (int i = 0; arr[i] != '\0'; i++) {
            printf("%d ", arr[i]); 
        }
        printf("\n");
    }
    else {
        wait(NULL);
        descending_order(arr);
        printf("Sorted array in descending order: ");
        for (int i = 0; arr[i] != '\0'; i++) {      
            printf("%d ", arr[i]);
        }
        printf("\n");
    }
    return 0;
}
